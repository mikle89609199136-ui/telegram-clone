FROM node:18-alpine

WORKDIR /app

# Копирование package.json и package-lock.json (если есть)
COPY package*.json ./

# Установка только production-зависимостей (не требует package-lock.json)
RUN npm install --production

# Копирование остальных файлов проекта
COPY . .

# Создание необходимых папок для данных и загрузок
RUN mkdir -p uploads logs data

# Открытие порта
EXPOSE 3000

# Команда запуска
CMD ["npm", "start"]
